#!/bin/bash

while true; do
  clear
  echo "Menu"
  echo "1. Ajout utilisateur"
  echo "2. Suppression utilisateur"
  echo "3. Liste utilisateur"
  echo "4. Quitter"
  echo "Choisissez une option:"
  read option

  case $option in
    1)
      echo "Entrez le nom de l'utilisateur:"
      read username
      echo "Entrez le nom du groupe principal:"
      read group
      sudo useradd -m -g $group $username
      echo "Utilisateur ajouté."
      ;;
    2)
      echo "Entrez le nom de l'utilisateur à supprimer:"
      read username
      sudo userdel -r $username
      echo "Utilisateur supprimé."
      ;;
    3)
      echo "Liste des utilisateurs:"
      cut -d: -f1 /etc/passwd
      ;;
    4)
      echo "Fin du script."
      exit 0
      ;;
    *)
      echo "Option invalide, veuillez réessayer."
      ;;
  esac
  read -n 1 -s -r -p "Appuyez sur une touche pour continuer"
done
